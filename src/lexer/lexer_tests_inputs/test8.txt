type arrayOfInteger is array[100] integer;
type two-dimensional-array is array[100] arrayOfInteger;

routine max-of-array(lengthOfArray: integer, arr: arrayOfInteger) : integer is 
    var ans : integer is -1000000000;
    for i in 1 .. lengthOfArray loop 
        if arr[i] > ans then 
           ans := arr[i];
        end 
    end 
    return ans ;
end 

routine sum(numberOfArrays: integer, lengthOfArray: integer, arr: two-dimensional-array) : integer is 
    var ans : integer is 0;
    for i in 1 .. numberOfArrays loop 
        ans := ans + max-of-array(lengthOfArray, arr[i]);
    end 
    return ans ;
end 

var arr : two-dimensional-array;
var result : integer;

arr[1][1] := 3;
arr[1][2] := 7;
arr[1][3] := 2;

arr[2][1] := 5;
arr[2][2] := 1;
arr[2][3] := 9;

arr[3][1] := 8;
arr[3][2] := 4;
arr[3][3] := 6;

result := sum(3, 3, arr);
