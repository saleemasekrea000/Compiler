type arrayOfInteger is array[100] integer;
type two-dimensional-array is array[100] arrayOfInteger;

routine max-of-two-dimensional-array(numberOfArrays: integer, lengthOfArray: integer, arr: two-dimensional-array) : integer is 
    var ans : integer is -10000000000;
    for i in 1 .. numberOfArrays loop 
        for j in 1 .. lengthOfArray loop 
            if arr[i][j] > ans then 
                ans := arr[i][j] ;
            end
        end
    end 
    return ans;
end

var arr : two-dimensional-array;
arr[1][1] := 2;
arr[1][2] := 4;
arr[1][3] := 1;

arr[2][1] := 7;
arr[2][2] := 9;
arr[2][3] := 5;

arr[3][1] := 6;
arr[3][2] := 3;
arr[3][3] := 8;

var result : integer is max-of-two-dimensional-array(3, 3, arr);
