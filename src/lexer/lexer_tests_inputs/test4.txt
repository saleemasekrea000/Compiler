//Reverse a List
type arrayOfInteger is array [100] integer;

routine reverse_array(n: integer, arr: arrayOfInteger) : arrayOfInteger is
    var reversed: arrayOfInteger;
    for i in reverse 1 .. n loop
        reversed[n-i+1] := arr[i];
    end
    return reversed;
end

var arr: arrayOfInteger is [1,2,3,4,5];

var reversed_arr : arrayOfInteger is reverse_array(5,arr);
