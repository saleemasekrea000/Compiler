// sum of an array of real numbers 
type arrayOfReals is array[100] real;

routine sum_of_reals(n:integer, arr:arrayOfReals) : real is 
     var ans : real is 0.0
     for i in 1 .. n loop 
         real := real + arr[i];
     end 
     return ans ;
end 

var arr : arrayOfReals is [1.0, 2.5 , 1, 4.7, 6.8, 0, -6, -100];

var sum : real is sum_of_reals(8, arr);
